# 阶段05：采集归位与数据集平台重组 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development`（推荐）或 `superpowers:executing-plans` 按任务执行。本仓库当前工作树较脏，且用户已明确要求直接在 `master` 当前工作区工作；执行时只允许 `git add` 本计划列出的文件，禁止顺手整理无关改动。步骤使用 checkbox (`- [ ]`) 语法跟踪。

**Goal:** 将秒级采集入口从分析中心迁回数据中心，形成 `关注币种 / 秒级采集 / 数据库库存` 三段式数据中心；同时把分析中心的“研究平台”收口为“数据集平台”，只保留 `数据集构建 / 模型训练` 两个子页。

**Architecture:** 后端新增一层 `data_center_collection` 兼容 API，对底层 `research_platform` collection service 做语义转发；前端新增数据中心采集 API / WebSocket 别名封装，在数据中心挂载新的采集宿主页面，并把分析中心现有 `ResearchPlatformPanel` 改造成只处理 dataset / training 的“双页数据集平台”。

**Tech Stack:** FastAPI、现有 `AppContext.research_platform()` 服务、Vue 3、Axios API service、现有 WebSocket client、Node built-in test runner、pytest。

**Execution Constraints:**
- 不修改 `模型训练构思.txt`。
- 不重写 `research_platform` 底层 storage schema。
- 不新增静默 fallback、mock 成功路径或假数据。
- 不改用户环境、`.env`、本地数据库内容。

- [ ] **Step 3: 实现数据中心采集 API 文件，直接转发到底层 research platform collection service**

```python
from __future__ import annotations

from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel, Field

from app.core.app_context import AppContext, get_app_context


router = APIRouter(prefix='/api/data-center/collections', tags=['Data Center Collection'])


class DataCenterCollectionSessionCreateRequest(BaseModel):
    inst_id: str
    planned_duration_sec: int = Field(ge=1)
    trigger_mode: str
    trigger_note: str = ''
    sampling_policy_id: str
    integrity_policy_version: str
    collector_version: str
    feature_recipe_version: str
    book_channel: str
    source_config_hash: str = 'manual'


@router.get('/sessions')
async def list_collection_sessions(limit: int = 50, ctx: AppContext = Depends(get_app_context)):
    return {'sessions': ctx.research_platform().list_collection_sessions(limit=limit)}


@router.get('/sessions/{session_id}')
async def get_collection_session(session_id: str, ctx: AppContext = Depends(get_app_context)):
    session = ctx.research_platform().get_collection_session_detail(session_id)
    if session is None:
        raise HTTPException(status_code=404, detail='collection session not found')
    return {'session': session}


@router.post('/sessions')
async def start_collection_session(
    request: DataCenterCollectionSessionCreateRequest,
    ctx: AppContext = Depends(get_app_context),
):
    session = await ctx.research_platform().start_collection_session(request.model_dump())
    return {'session': session}


@router.post('/sessions/{session_id}/stop')
async def stop_collection_session(session_id: str, ctx: AppContext = Depends(get_app_context)):
    session = await ctx.research_platform().stop_collection_session(session_id)
    if session is None:
        raise HTTPException(status_code=404, detail='collection session not found')
    return {'session': session}


@router.get('/census/status')
async def get_collection_census_status(ctx: AppContext = Depends(get_app_context)):
    return {'status': ctx.research_platform().get_census_status()}
```

- [ ] **Step 5: 新建 `useDataCollectionWorkspace.mjs`，让数据中心采集页只依赖新的封装名**

```javascript
import { ref } from 'vue';

import { api } from '../services/api.js';
import marketWS from '../services/websocket.js';

export function useDataCollectionWorkspace() {
  const sessions = ref([]);
  const activeSession = ref(null);
  const selectedSessionId = ref('');
  const sessionCharts = ref({ price: [], trade: [], book: [] });
  const sessionCoverage = ref({
    coverageRatio: 0,
    validSecondCount: 0,
    missingSecondCount: 0,
    bookStaleSecondCount: 0,
    stateStaleSecondCount: 0,
    inferenceReadyCount: 0,
    trainingReadyCount: 0,
    stratumCoverage: [],
  });
  const censusStatus = ref(null);

  const loadSessions = async () => {
    const response = await api.getDataCenterCollectionSessions({ limit: 50 });
    sessions.value = response.sessions || [];
    if (!selectedSessionId.value && sessions.value.length > 0) {
      selectedSessionId.value = sessions.value[0].session_id;
    }
  };

  const loadSessionDetail = async (sessionId) => {
    if (!sessionId) {
      activeSession.value = null;
      sessionCharts.value = { price: [], trade: [], book: [] };
      sessionCoverage.value = {
        coverageRatio: 0,
        validSecondCount: 0,
        missingSecondCount: 0,
        bookStaleSecondCount: 0,
        stateStaleSecondCount: 0,
        inferenceReadyCount: 0,
        trainingReadyCount: 0,
        stratumCoverage: [],
      };
      return;
    }

    const response = await api.getDataCenterCollectionSessionDetail(sessionId);
    activeSession.value = response.session || null;
    sessionCharts.value = response.session?.charts || { price: [], trade: [], book: [] };
    sessionCoverage.value = normalizeCoverage(response.session?.coverage || {});
  };

  const loadCensusStatus = async () => {
    const response = await api.getDataCenterCollectionCensusStatus();
    censusStatus.value = response.status || null;
  };

  const handleCollectionEvent = async (payload = {}) => {
    if (payload.event === 'session_updated') {
      await loadSessions();
      if (payload.session_id) {
        selectedSessionId.value = payload.session_id;
        await loadSessionDetail(payload.session_id);
      }
    }
    if (payload.event === 'census_updated') {
      await loadCensusStatus();
    }
  };

  const attachRealtime = () => {
    marketWS.connect().catch(() => {});
    marketWS.subscribeDataCenterCollection(handleCollectionEvent);
  };

  const detachRealtime = () => {
    marketWS.unsubscribeDataCenterCollection(handleCollectionEvent);
  };

  return {
    sessions,
    activeSession,
    selectedSessionId,
    sessionCharts,
    sessionCoverage,
    censusStatus,
    loadSessions,
    loadSessionDetail,
    loadCensusStatus,
    attachRealtime,
    detachRealtime,
  };
}

function normalizeCoverage(coverage = {}) {
  return {
    coverageRatio: coverage.coverage_ratio ?? coverage.coverageRatio ?? 0,
    validSecondCount: coverage.valid_second_count ?? coverage.validSecondCount ?? 0,
    missingSecondCount: coverage.missing_second_count ?? coverage.missingSecondCount ?? 0,
    bookStaleSecondCount: coverage.book_stale_second_count ?? coverage.bookStaleSecondCount ?? 0,
    stateStaleSecondCount: coverage.state_stale_second_count ?? coverage.stateStaleSecondCount ?? 0,
    inferenceReadyCount: coverage.inference_ready_count ?? coverage.inferenceReadyCount ?? 0,
    trainingReadyCount: coverage.training_ready_count ?? coverage.trainingReadyCount ?? 0,
    stratumCoverage: coverage.stratum_coverage ?? coverage.stratumCoverage ?? [],
  };
}
```
