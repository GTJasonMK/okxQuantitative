- [ ] **Step 3: 实现动作、`sessionProgress` 状态与细事件归一化**

```javascript
export function useDataCollectionWorkspace(deps = {}) {
  const api = deps.api ?? dataCenterApi;
  const sessionProgress = ref(null);

  const normalizeProgress = progress => ({
    writtenSeconds: Number(progress?.written_seconds || 0),
    remainingSeconds: Number(progress?.remaining_seconds || 0),
    secondsToFullWindow: Number(progress?.seconds_to_full_window || 0),
    secondsToNextBoundary: Number(progress?.seconds_to_next_boundary || 0),
  });

  const startCollectionSession = async payload => {
    const response = await api.createDataCenterCollectionSession(payload);
    await loadSessions();
    selectedSessionId.value = response.session?.session_id || '';
    await loadSessionDetail(selectedSessionId.value);
  };

  const stopCollectionSession = async sessionId => {
    await api.stopDataCenterCollectionSession(sessionId);
    await loadSessions();
    await loadSessionDetail(sessionId);
  };

  const loadSessionDetail = async sessionId => {
    const response = await api.getDataCenterCollectionSession(sessionId);
    activeSession.value = response.session || null;
    sessionCoverage.value = normalizeCoverage(response.session?.coverage);
    sessionProgress.value = normalizeProgress(response.session?.progress);
    sessionCharts.value = normalizeCharts(response.session?.charts);
  };

  const handleCollectionEvent = async (payload = {}) => {
    if (!payload.session_id) {
      return;
    }
    if (payload.event === 'second_flushed' || payload.event === 'session_quality_updated') {
      if (payload.session_id === selectedSessionId.value) {
        await loadSessionDetail(payload.session_id);
      }
      return;
    }
    if (payload.event?.startsWith('session_')) {
      await refreshSessionSelection(payload, selectedSessionId, loadSessions, loadSessionDetail);
    }
  };

  return {
    selectedSessionId,
    sessionProgress,
    startCollectionSession,
    stopCollectionSession,
    handleCollectionEvent,
  };
}
```
